function actualizarInterfaz(carrito) { const tbody = document.querySelector('#tablaResumen tbody'); tbody.innerHTML = ""; // Limpiamos la tabla antes de renderizar if (!carrito || carrito.length === 0) return; carrito.forEach(item => { // La resta ya viene calculada del servidor en item.precioVenta const fila = ` ${item.nombre} $${parseFloat(item.precioVenta).toFixed(2)} ${item.descuento > 0 ? `
(-${item.descuento}%)` : ''} ${item.cantidad} $${parseFloat(item.subtotal).toFixed(2)} `; tbody.innerHTML += fila; }); // Calculamos el Gran Total sumando los subtotales ya restados const totalFinal = carrito.reduce((acc, i) => acc + parseFloat(i.subtotal), 0); document.getElementById('granTotal').innerText = `TOTAL: $${totalFinal.toFixed(2)}`; document.getElementById('resumenVenta').style.display = 'block'; } // Asegúrate de que tu función procesarVenta llame a actualizarInterfaz al final async function procesarVenta() { const filas = document.querySelectorAll('.fila-producto'); // Opcional: Podrías añadir un fetch para limpiar el carrito aquí si quieres fresh start for (let fila of filas) { const idProducto = fila.querySelector('.idProducto').value; const cantidad = fila.querySelector('.cantidad').value; const codigoPromo = fila.querySelector('.promo').value; if (idProducto) { await fetch('/ventas/carrito', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ idProducto, cantidad, codigoPromo }) }); } } // Obtener los datos actualizados del servidor const res = await fetch('/ventas/carrito'); const data = await res.json(); actualizarInterfaz(data.body); }
